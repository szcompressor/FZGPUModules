239 void deserializeHeader(const uint8_t* header_buffer, size_t size) override {
240 // Minimum size is the original 32-byte layout (before user_eb/value_base were added).
241 constexpr size_t kLegacySize = 32;
242 if (size < kLegacySize) {
243 throw std::runtime_error("Invalid Lorenzo config size");
244 }
245
246 LorenzoQuantConfig config;
247 std::memcpy(&config, header_buffer, std::min(size, sizeof(LorenzoQuantConfig)));
248
249 // error_bound in the header is always the absolute bound used at compression.
250 config_.error_bound = config.error_bound;
251 computed_abs_eb_ = static_cast<TInput>(config.error_bound);
252 config_.quant_radius = static_cast<TCode>(config.quant_radius);
253 num_elements_ = config.num_elements;
254 actual_outlier_count_= config.outlier_count;
255 // New fields: present only in headers written by v1+ (≥40B, added user_eb/value_base/eb_mode).
256 constexpr size_t kV1Size = 40;
257 if (size >= kV1Size) {
258 config_.eb_mode = static_cast<ErrorBoundMode>(config.eb_mode);
259 config_.precomputed_value_base = config.value_base;
260 computed_value_base_ = config.value_base;
261 } else {
262 config_.eb_mode = ErrorBoundMode::ABS;
263 config_.precomputed_value_base = 0.0f;
264 computed_value_base_ = 0.0f;
265 }
266 // zigzag_codes field added in v2 (≥44B).
267 if (size >= sizeof(LorenzoQuantConfig)) {
268 config_.zigzag_codes = (config.zigzag_codes != 0);
269 } else {
270 config_.zigzag_codes = false;
271 }
272
273 // Restore spatial dimensions; handle old (pre-dims) files gracefully
274 int eff_ndim = (config.ndim == 0) ? 1 : static_cast<int>(config.ndim);
275 // dim_x: stored explicitly; fall back to derivation for old files
276 if (config.dim_x > 0) {
277 config_.dims[0] = config.dim_x;
278 } else if (config.num_elements > 0) {
279 size_t yz = std::max<size_t>(1, config.dim_y) * std::max<size_t>(1, config.dim_z);
280 config_.dims[0] = config.num_elements / yz;
281 } else {
282 config_.dims[0] = 0;
283 }
284 if (eff_ndim >= 2) {
285 config_.dims[1] = (config.dim_y > 0) ? config.dim_y : 1;
286 } else {
287 config_.dims[1] = 1;
288 }
289 if (eff_ndim >= 3) {
290 config_.dims[2] = (config.dim_z > 0) ? config.dim_z : 1;
291 } else {
292 config_.dims[2] = 1;
293 }
294 }
