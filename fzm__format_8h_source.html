277inline size_t getDataTypeSize(DataType type) {
278 switch (type) {
279 case DataType::UINT8: case DataType::INT8: return 1;
280 case DataType::UINT16: case DataType::INT16: return 2;
281 case DataType::UINT32: case DataType::INT32: case DataType::FLOAT32: return 4;
282 case DataType::UINT64: case DataType::INT64: case DataType::FLOAT64: return 8;
283 default: throw std::runtime_error("Unknown data type");
