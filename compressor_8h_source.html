644template<typename StageT, typename... Args>
645StageT* Pipeline::addStage(Args&&... args) {
646 if (is_finalized_) {
647 throw std::runtime_error("Cannot add stages after finalization");
648 }
649
650 auto stage_ptr = std::make_unique<StageT>(std::forward<Args>(args)...);
651 StageT* stage = stage_ptr.get();
652
653 stage->setDims(dims_);
654
655 DAGNode* node = dag_->addStage(stage, stage->getName());
656 size_t num_outputs = stage->getNumOutputs();
657 auto output_names = stage->getOutputNames();
658
659 // Pre-allocate all output slots as unconnected (size=1 placeholder).
660 // connect() will promote any that get wired to downstream stages.
661 for (size_t i = 0; i < num_outputs; i++) {
662 std::string out_name = i < output_names.size() ? output_names[i] : std::to_string(i);
663 dag_->addUnconnectedOutput(node, 1, i, stage->getName() + "." + out_name + "_unconnected");
664 }
665
666 stage_to_node_[stage] = node;
667 stages_.push_back(std::move(stage_ptr));
668 return stage;
669}
