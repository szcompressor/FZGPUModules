609template<typename StageT, typename... Args>
610StageT* Pipeline::addStage(Args&&... args) {
611 if (is_finalized_) {
612 throw std::runtime_error("Cannot add stages after finalization");
613 }
614
615 auto stage_ptr = std::make_unique<StageT>(std::forward<Args>(args)...);
616 StageT* stage = stage_ptr.get();
617
618 DAGNode* node = dag_->addStage(stage, stage->getName());
619 size_t num_outputs = stage->getNumOutputs();
620 auto output_names = stage->getOutputNames();
621
622 // Pre-allocate all output slots as unconnected (size=1 placeholder).
623 // connect() will promote any that get wired to downstream stages.
624 for (size_t i = 0; i < num_outputs; i++) {
625 std::string out_name = i < output_names.size() ? output_names[i] : std::to_string(i);
626 dag_->addUnconnectedOutput(node, 1, i, stage->getName() + "." + out_name + "_unconnected");
627 }
628
629 stage_to_node_[stage] = node;
630 stages_.push_back(std::move(stage_ptr));
631 return stage;
632}
