99 if (input_sizes.empty()) return {0};
100 if (!is_inverse_) {
101 // Forward: packed output is ceil(n * nbits / 8) bytes.
102 const size_t n = input_sizes[0] / sizeof(T);
103 return {(n * nbits_ + 7) / 8};
104 } else {
105 // Inverse: worst case — every packed bit expands to a full element.
106 // input_sizes[0] is the packed byte count; max elements = bytes * (8/nbits).
107 const size_t max_elems = (input_sizes[0] * 8 + nbits_ - 1) / nbits_;
108 return {max_elems * sizeof(T)};
109 }
